#include <stdio.h>
#define PLUS 900

int n, in[101];
long long dp[101][21];
bool check[101][21];

long long f(int x, int y){
    if(y<0 || y>20) return 0;
    if(x == 1){
        if(y == in[1]) return 1;
        else return 0;
    }

    if(check[x][y] == false){
        check[x][y] = true;
        dp[x][y] = f(x-1, y+in[x]) + f(x-1, y-in[x]);
    }
    return dp[x][y];
}

int main(){
    scanf("%d", &n);
    for(int i=1; i<=n; i++) scanf("%d", &in[i]);

    long long out = 0;
    printf("%lld", f(n-1, in[n]));
}
