#include <stdio.h>
#include <algorithm>
#define INF 2147483647

using namespace std; 
int n;
pair<int,int> in[501];
int dp[501][501];
bool check[501][501];

int f(int x, int y){
    if(x==y) return 0;

    if(check[x][y] == false){
        check[x][y] = true;
        dp[x][y] = INF;
        for(int i=x; i<=y-1; i++){
            dp[x][y] = min(dp[x][y], f(x,i)+f(i+1,y) + in[x].first*in[i].second*in[y].second);
        }
    }
    return dp[x][y];
}

int main(){
    scanf("%d", &n);
    for(int i=1; i<=n; i++){
        scanf("%d %d", &in[i].first, &in[i].second);
    }

    printf("%d", f(1,n));
}
